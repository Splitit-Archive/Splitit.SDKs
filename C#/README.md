## Flex Fields

Commong usage for .NET SDK is in making necessary server-side requests as part of FlexFields product integration.
The code below is an example of how SDK wrappers can be used to simplify acquiring public token and verifying payment.
For more information, please visit [FlexFields documentation](https://hosted.production.splitit.com/#netcoremvc).

```csharp
using System;
using System.Diagnostics;
using Splitit.SDK.Client.Api;
using Splitit.SDK.Client.Client;
using Splitit.SDK.Client.Model;

namespace Example
{
    public class FlexFieldsIntegrationController : Controller
    {
        public ActionResult Initialize()
        {
            Configuration.Sandbox.AddApiKey("_YOUR_API_KEY_");

            var publicToken = FlexFields.Authenticate(Configuration.Sandbox, "_YOUR_USERNAME_", "_YOUR_PASSWORD_")
                .GetPublicToken(1000, "USD");

            return Json({ token: publicToken });
        }

        public ActionResult VerifyPayment(string planNumber)
        {
            // fetch order from DB here
            Configuration.Sandbox.AddApiKey("_YOUR_API_KEY_");

            var ok = FlexFields.Authenticate(Configuration.Sandbox, "_YOUR_USERNAME_", "_YOUR_PASSWORD_")
                .VerifyPayment(planNumber, 1000);

            if (!ok){
                return BadResult();
            } else {
                return Ok();
            }
        }

    }
}
```
